create table public.employees (
  id uuid not null default gen_random_uuid (),
  employee_code character varying(20) not null,
  pn character varying(20) null,
  firstname character varying(100) null,
  lastname character varying(100) null,
  fullname character varying(200) null,
  eng_name character varying(200) null,
  lao_name character varying(200) null,
  dob date null,
  age integer null,
  ssno character varying(50) null,
  tel character varying(50) null,
  address text null,
  village character varying(100) null,
  district character varying(100) null,
  province character varying(100) null,
  position character varying(200) null,
  department character varying(100) null,
  project character varying(100) null,
  cost character varying(50) null,
  company character varying(200) null,
  group_type character varying(50) null,
  type_of_work character varying(10) null,
  status character varying(50) null,
  start_date date null,
  end_date date null,
  real_resigned date null,
  experience character varying(50) null,
  pa45d date null,
  pa90d date null,
  pa119d date null,
  created_at timestamp with time zone null default now(),
  updated_at timestamp with time zone null default now(),
  constraint employees_pkey primary key (id),
  constraint employees_employee_code_key unique (employee_code)
) TABLESPACE pg_default;

create index IF not exists idx_employees_employee_code on public.employees using btree (employee_code) TABLESPACE pg_default;

create index IF not exists idx_employees_department on public.employees using btree (department) TABLESPACE pg_default;

create index IF not exists idx_employees_status on public.employees using btree (status) TABLESPACE pg_default;

create trigger trg_employees_updated_at BEFORE
update on employees for EACH row
execute FUNCTION update_updated_at ();



CREATE TABLE courses (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    course_name TEXT NOT NULL,
   created_by TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
); 

CREATE TABLE system_users (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    emp_code TEXT UNIQUE NOT NULL,
    fullname TEXT NOT NULL,
    position TEXT,
    department TEXT,
    username TEXT UNIQUE NOT NULL,
    password TEXT NOT NULL,
    role TEXT NOT NULL DEFAULT 'user',
    created_by TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

create table public.employee_training_records (
     id              UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    "group"         text,

    first_name      text,

    position        text,                       
    department      text,                       
    gender          text,                       
    nationality     text,                       
    status          text,                       
    course_name     text,

CREATE TABLE public.user_logs (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  system_user_id uuid NOT NULL,
  user_agent text,
  ip_address inet,
  action text NOT NULL,
  old_value jsonb,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  view_mode text,
  CONSTRAINT user_logs_pkey PRIMARY KEY (id),
  CONSTRAINT user_logs_system_user_id_fkey FOREIGN KEY (system_user_id) REFERENCES public.system_users(id)
);
